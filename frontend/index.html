快速开始

安装依赖

确保你有 C++20 编译器和 CMake 3.16+

02

克隆仓库

获取 galay-kernel 和 galay-http 源码

03

编译安装

使用 CMake 构建并安装到系统

04

开始使用

在你的项目中引入 galay 库

# 克隆仓库
git clone https://github.com/galay/galay-kernel.git
git clone https://github.com/galay/galay-http.git

# 编译 galay-kernel
cd galay-kernel
mkdir build && cd build
cmake .. -DCMAKE_BUILD_TYPE=Release
make -j$(nproc)
sudo make install

# 编译 galay-http
cd ../../galay-http
mkdir build && cd build
cmake .. -DCMAKE_BUILD_TYPE=Release
make -j$(nproc)
sudo make install
#include "galay-http/kernel/http/HttpServer.h"
#include "galay-http/kernel/http/HttpRouter.h"

using namespace galay::http;

Coroutine helloHandler(HttpConn& conn, HttpRequest req) {
    auto response = Http1_1ResponseBuilder::ok()
        .text("Hello, Galay!")
        .build();

    auto writer = conn.getWriter();
    while (true) {
        auto result = co_await writer.sendResponse(response);
        if (!result || result.value()) break;
    }
}

int main() {
    HttpRouter router;
    router.addHandler<HttpMethod::GET>("/", helloHandler);

    HttpServer server({"0.0.0.0", 8080});
    server.start(std::move(router));
}
